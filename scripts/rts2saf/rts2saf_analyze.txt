                          [--topath PATH] [--logfile LOGFILE] [--toconsole]

  --topath PATH         : ., write log file to path

  --toconsole           : False, log to console
  --config CONFIG       : /etc/rts2/rts2saf/rts2saf.cfg, configuration file
                        path
  --basepath BASEPATH   : None, directory where FITS images from possibly many
                        focus runs are stored
  --filternames FILTERNAMES 
                        : None, list of filters to analyzed, None: all
  --displayds9          : False, display fits images and region files
  --displayfit          : False, display fit
  --cataloganalysis     : False, ananlys is done with CatalogAnalysis
